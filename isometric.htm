function getSnappedPoint(x1, y1, x2, y2) { const dx = x2 - x1; const dy = y2 - y1; const dist = Math.sqrt(dx * dx + dy * dy); // Если палец почти не сдвинулся, не рисуем if (dist < 10) return {x: x1, y: y1}; // Получаем угол в радианах и переводим в градусы (от -180 до 180) let angleDeg = Math.atan2(dy, dx) * 180 / Math.PI; // Массив всех направлений (полный круг 360 градусов) // Добавляем 0, 30, 90, 150, 180, -150, -90, -30 const isoAngles = [0, 30, 90, 150, 180, -150, -90, -30]; // Находим ближайший угол из списка const closestAngle = isoAngles.reduce((prev, curr) => { return (Math.abs(curr - angleDeg) < Math.abs(prev - angleDeg) ? curr : prev); }); // Пересчитываем финальные координаты const finalRad = closestAngle * Math.PI / 180; return { x: x1 + dist * Math.cos(finalRad), y: y1 + dist * Math.sin(finalRad) }; }
