const canvas = document.getElementById('paintCanvas'); const ctx = canvas.getContext('2d'); let lines = []; let startX, startY, currentX, currentY; let isDrawing = false; // 6 направлений изометрии + горизонталь const isoAngles = [0, 30, 90, 150, 180, -150, -90, -30]; function init() { canvas.width = window.innerWidth; canvas.height = window.innerHeight; render(); } function drawGrid() { ctx.strokeStyle = '#e0e0e0'; ctx.lineWidth = 1; const spacing = 40; for (let x = -canvas.width; x < canvas.width * 2; x += spacing) { ctx.beginPath(); ctx.moveTo(x, 0); ctx.lineTo(x, canvas.height); ctx.stroke(); ctx.beginPath(); ctx.moveTo(x, 0); ctx.lineTo(x + canvas.height * Math.tan(30 * Math.PI / 180), canvas.height); ctx.stroke(); ctx.beginPath(); ctx.moveTo(x, 0); ctx.lineTo(x - canvas.height * Math.tan(30 * Math.PI / 180), canvas.height); ctx.stroke(); } } function getSnappedPoint(x1, y1, x2, y2) { const dx = x2 - x1; const dy = y2 - y1; const dist = Math.sqrt(dx * dx + dy * dy); if (dist < 10) return {x: x1, y: y1}; let angleDeg = Math.atan2(dy, dx) * 180 / Math.PI; const closestAngle = isoAngles.reduce((prev, curr) => { return (Math.abs(curr - angleDeg) < Math.abs(prev - angleDeg) ? curr : prev); }); const finalRad = closestAngle * Math.PI / 180; return { x: x1 + dist * Math.cos(finalRad), y: y1 + dist * Math.sin(finalRad) }; } function render() { ctx.clearRect(0, 0, canvas.width, canvas.height); drawGrid(); ctx.lineWidth = 3; ctx.lineCap = 'round'; lines.forEach(line => { ctx.strokeStyle = '#2c3e50'; ctx.beginPath(); ctx.moveTo(line.x1, line.y1); ctx.lineTo(line.x2, line.y2); ctx.stroke(); }); if (isDrawing) { const snapped = getSnappedPoint(startX, startY, currentX, currentY); ctx.strokeStyle = '#3498db'; ctx.beginPath(); ctx.moveTo(startX, startY); ctx.lineTo(snapped.x, snapped.y); ctx.stroke(); } } // --- УНИВЕРСАЛЬНЫЕ ОБРАБОТЧИКИ --- function startDrawing(x, y) { isDrawing = true; startX = x; startY = y; currentX = x; currentY = y; } function moveDrawing(x, y) { if (!isDrawing) return; currentX = x; currentY = y; render(); } function stopDrawing() { if (isDrawing) { const snapped = getSnappedPoint(startX, startY, currentX, currentY); if (Math.abs(snapped.x - startX) > 5 || Math.abs(snapped.y - startY) > 5) { lines.push({x1: startX, y1: startY, x2: snapped.x, y2: snapped.y}); } isDrawing = false; render(); } } // Мышь canvas.addEventListener('mousedown', e => startDrawing(e.clientX, e.clientY)); window.addEventListener('mousemove', e => moveDrawing(e.clientX, e.clientY)); window.addEventListener('mouseup', stopDrawing); // Тач (телефон) canvas.addEventListener('touchstart', e => { const t = e.touches[0]; startDrawing(t.clientX, t.clientY); }); canvas.addEventListener('touchmove', e => { const t = e.touches[0]; moveDrawing(t.clientX, t.clientY); }); canvas.addEventListener('touchend', stopDrawing); // Горячие клавиши (Undo на Ctrl+Z) window.addEventListener('keydown', e => { if (e.ctrlKey && e.code === 'KeyZ') undo(); }); function undo() { lines.pop(); render(); } function clearCanvas() { if(confirm("Очистить весь рисунок?")) { lines = []; render(); } } window.addEventListener('resize', init); init();
